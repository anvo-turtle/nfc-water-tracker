    <h1>Water Intake Tracker</h1>

    <p>Current intake: <span id="count">0</span> units</p>

    <button id="increment" onclick="increment()">+1</button>

    <button id="decrement" onclick="decrement()">-1</button>


    <script>

        let count = parseInt(localStorage.getItem('waterCount')) || 0;


        function updateDisplay() {

            document.getElementById('count').innerText = count;

        }


        function increment() {

            count++;

            localStorage.setItem('waterCount', count);

            updateDisplay();

        }


        function decrement() {

            if (count > 0) {

                count--;

                localStorage.setItem('waterCount', count);

                updateDisplay();

            }

        }


        // Automatically increment on page load (for NFC scan or visit)

        increment();

    </script>
